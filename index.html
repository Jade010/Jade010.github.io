import datetime

class Jade:
    def __init__(self):
        self.name = "Jade Aidoghie"
        self.profession = "Software Developer"
        self.education = "BS in Computer Science"
        self.skills = ["Python", "JavaScript", "HTML", "CSS"]
        self.hobbies = ["Photography", "Hiking", "Gaming"]
    
    def get_bio(self):
        return f"My name is {self.name}, a {self.profession}. I have a {self.education} and my skills include " + ", ".join(self.skills) + "."

# Create an instance of Jade
person = Jade()
print(person.get_bio())
print("Hobbies include: " + ", ".join(person.hobbies))
